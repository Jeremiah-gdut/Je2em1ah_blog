def decrypt2(n: int, c: int, hint: int):

  b = pow(hint, -1, n)
  c_quadratic = -514 * n

  delta = b * b - 4 * 114 * c_quadratic

  sqrt_delta = int(math.isqrt(delta))

  q = (-b + sqrt_delta) // (2 * 114)

  if n % q != 0:
      q = (-b - sqrt_delta) // (2 * 114)

  p = n // q

  phi = (p - 1) * (q - 1)
  d = pow(e, -1, phi)

  m = pow(c, d, n)
  return long_to_bytes(m).decode()
