def boneh_durfee(pol, modulus, mm, tt, XX, YY):
    """
    Boneh and Durfee revisited by Herrmann and May
 
 Âú®‰ª•‰∏ãÊÉÖÂÜµ‰∏ãÊâæÂà∞Ëß£ÂÜ≥ÊñπÊ°àÔºö
* d < N^delta
* |x|< e^delta
* |y|< e^0.5
ÊØèÂΩì delta < 1 - sqrtÔºà2Ôºâ/2 ~ 0.292
    """
 
    # substitution (Herrman and May)
    PR.<u, x, y> = PolynomialRing(ZZ)   #Â§öÈ°πÂºèÁéØ
    Q = PR.quotient(x*y + 1 - u)        #  u = xy + 1
    polZ = Q(pol).lift()
 
    UU = XX*YY + 1
 
    # x-Áßª‰Ωç
    gg = []
    for kk in range(mm + 1):
        for ii in range(mm - kk + 1):
            xshift = x^ii * modulus^(mm - kk) * polZ(u, x, y)^kk
            gg.append(xshift)
    gg.sort()
 
    # ÂçïÈ°πÂºè x Áßª‰ΩçÂàóË°®
    monomials = []
    for polynomial in gg:
        for monomial in polynomial.monomials(): #ÂØπ‰∫éÂ§öÈ°πÂºè‰∏≠ÁöÑÂçïÈ°πÂºè„ÄÇÂçïÈ°πÂºèÔºàÔºâÔºö
            if monomial not in monomials:  # Â¶ÇÊûúÂçïÈ°π‰∏çÂú®ÂçïÈ°π‰∏≠
                monomials.append(monomial)
    monomials.sort()
 
    # y-Áßª‰Ωç
    for jj in range(1, tt + 1):
        for kk in range(floor(mm/tt) * jj, mm + 1):
            yshift = y^jj * polZ(u, x, y)^kk * modulus^(mm - kk)
            yshift = Q(yshift).lift()
            gg.append(yshift) # substitution
 
    # ÂçïÈ°πÂºè y Áßª‰ΩçÂàóË°®
    for jj in range(1, tt + 1):
        for kk in range(floor(mm/tt) * jj, mm + 1):
            monomials.append(u^kk * y^jj)
 
    # ÊûÑÈÄ†Ê†º B
    nn = len(monomials)
    BB = Matrix(ZZ, nn)
    for ii in range(nn):
        BB[ii, 0] = gg[ii](0, 0, 0)
        for jj in range(1, ii + 1):
            if monomials[jj] in gg[ii].monomials():
                BB[ii, jj] = gg[ii].monomial_coefficient(monomials[jj]) * monomials[jj](UU,XX,YY)
 
    #Á∫¶ÂåñÊ†ºÁöÑÂéüÂûã
    if helpful_only:

        BB = remove_unhelpful(BB, monomials, modulus^mm, nn-1)
        # ÈáçÁΩÆÁª¥Â∫¶
        nn = BB.dimensions()[0]
        if nn == 0:
            print ("failure")
            return 0,0
 
    # Ê£ÄÊü•ÂêëÈáèÊòØÂê¶ÊúâÂ∏ÆÂä©
    if debug:
        helpful_vectors(BB, modulus^mm)
 
    # Ê£ÄÊü•Ë°åÂàóÂºèÊòØÂê¶Ê≠£Á°ÆÁïåÂÆö
    det = BB.det()
    bound = modulus^(mm*nn)
    if det >= bound:
        print ("We do not have det < bound. Solutions might not be found.")
        print ("Try with highers m and t.")
        if debug:
            diff = (log(det) - log(bound)) / log(2)
            print ("size det(L) - size e^(m*n) = ", floor(diff))
        if strict:
            return -1, -1
    else:
        print ("det(L) < e^(m*n) (good! If a solution exists < N^delta, it will be found)")

    if debug:
        matrix_overview(BB, modulus^mm)

    if debug:
